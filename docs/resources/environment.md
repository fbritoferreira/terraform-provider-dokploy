# dokploy_environment (Resource)

Manages an environment within a Dokploy project. Environments allow you to organize and isolate applications, databases, and other resources within a project, enabling workflows like staging/production separation.

## Example Usage

### Basic Environment

```terraform
resource "dokploy_project" "myproject" {
  name        = "My Application"
  description = "Main application project"
}

resource "dokploy_environment" "production" {
  name        = "Production"
  description = "Production environment"
  project_id  = dokploy_project.myproject.id
}
```

### Multiple Environments for CI/CD

```terraform
resource "dokploy_project" "webapp" {
  name        = "Web Application"
  description = "Main web application"
}

resource "dokploy_environment" "development" {
  name        = "Development"
  description = "Development environment for testing new features"
  project_id  = dokploy_project.webapp.id
}

resource "dokploy_environment" "staging" {
  name        = "Staging"
  description = "Staging environment for pre-production testing"
  project_id  = dokploy_project.webapp.id
}

resource "dokploy_environment" "production" {
  name        = "Production"
  description = "Production environment serving live traffic"
  project_id  = dokploy_project.webapp.id
}
```

### Environment with Applications

```terraform
resource "dokploy_project" "api" {
  name = "API Services"
}

resource "dokploy_environment" "staging" {
  name        = "Staging"
  description = "Staging environment"
  project_id  = dokploy_project.api.id
}

resource "dokploy_application" "api_staging" {
  name           = "api-staging"
  project_id     = dokploy_project.api.id
  environment_id = dokploy_environment.staging.id
  source_type    = "docker"
  docker_image   = "myregistry/api:staging"
}

resource "dokploy_database" "postgres_staging" {
  name            = "postgres-staging"
  project_id      = dokploy_project.api.id
  environment_id  = dokploy_environment.staging.id
  type            = "postgres"
  docker_image    = "postgres:16"
  database_name   = "api_staging"
  database_user   = "api"
  database_password = "staging-password"
}
```
